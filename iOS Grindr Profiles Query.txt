--iOS Grindr Profiles Query v24.x and 25.2. 25.2 has much more data, this needs updated to include enrichments, espescially from zprofileentity.

SELECT
    ZINBOXCONVERSATIONENTITY.ZNAME AS 'Username',
    ZINBOXCONVERSATIONPARTICIPANTENTITY.ZPROFILEID AS 'Profile ID',
    ZINBOXCONVERSATIONENTITY.ZCONVERSATIONID AS 'Conversation ID',
    datetime('2001-01-01', "ZLASTMESSAGETIMESTAMP" || ' seconds') AS 'Last Message Time',
    ZINBOXCONVERSATIONENTITY.ZUNREADCOUNT AS 'Unread Messages',
    ZINBOXCONVERSATIONPARTICIPANTENTITY.ZRAWDISTANCE AS 'Distance (Meters)',
    ZINBOXCONVERSATIONPARTICIPANTENTITY.ZPRIMARYMEDIAHASH AS 'Profile Media Hash'
FROM
    ZINBOXCONVERSATIONENTITY
JOIN
    ZINBOXCONVERSATIONPARTICIPANTENTITY
ON
    ZINBOXCONVERSATIONPARTICIPANTENTITY.ZINBOXCONVERSATION = ZINBOXCONVERSATIONENTITY.Z_PK
ORDER by ZINBOXCONVERSATIONENTITY.Z_PK asc;
